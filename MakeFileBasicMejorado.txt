# Toolchain
CC      = gcc
CFLAGS  = -std=c17 -Wall -Wextra -Wpedantic -O2 -I./dat -I./archivos

# Proyecto
TARGET  = myprogram
SRCS    = main.c

# Dirs
BINDIR  = build
OBJDIR  = build

# Extensión de ejecutable según SO
ifeq ($(OS),Windows_NT)
  EXEEXT := .exe
else
  EXEEXT :=
endif

# Derivados
OBJS         := $(SRCS:%.c=$(OBJDIR)/%.o)
TARGET_PATH  := $(BINDIR)/$(TARGET)$(EXEEXT)

.PHONY: all run clean mrproper

all: $(TARGET_PATH)

run: $(TARGET_PATH)
	./$(TARGET_PATH)

# Link
$(TARGET_PATH): $(OBJS) | $(BINDIR)
	$(CC) $(CFLAGS) $(OBJS) -o $@

# Compile  (ojo: %.o con letra "o", no cero)
$(OBJDIR)/%.o: %.c | $(OBJDIR)
	$(CC) $(CFLAGS) -c $< -o $@

# Crear carpetas si no existen
$(BINDIR) $(OBJDIR):
	mkdir -p $@

# Limpieza: borra objetos y ejecutables en build/
clean:
	rm -f $(OBJDIR)/*.o $(TARGET_PATH) $(BINDIR)/*.exe $(BINDIR)/*.out $(BINDIR)/*.bin

# Limpieza total: borra la carpeta build
mrproper: clean
	rm -rf $(BINDIR)


#make → compila a build/myprogram[...]
#make run → ejecuta desde build/
#make clean → borra objetos y ejecutables dentro de build/
#make mrproper → borra toda la carpeta build/
